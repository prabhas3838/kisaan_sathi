const YahooFinance = require('yahoo-finance2').default;
const yahooFinance = new YahooFinance({ suppressNotices: ['ripHistorical'] });
const MandiPrice = require("../models/MandiPrice");
 
// Dictionary to map generic crops to their global Yahoo Finance commodity ticker
const TICKER_MAP = {
    "wheat": "ZW=F",    // Wheat Futures
    "tomato": "ZS=F",   // Soybean pattern used as proxy if specific veg isn't on global exchange
    "corn": "ZC=F",     // Corn Futures
    "rice": "ZR=F"      // Rough Rice Futures
};
 
/**
 * Service to fetch historical crop prices from an external live API.
 * Uses Yahoo Finance for Keyless Global Commodity Data instead of Govt APIs.
 */
exports.fetchLiveHistoricalData = async (crop, mandi) => {
    try {
        const ticker = TICKER_MAP[crop.toLowerCase()];
 
        if (!ticker) {
            console.log(`No global ticker found for ${crop}, falling back to local simulation data.`);
            return populateFallbackData(crop, mandi);
        }
 
        // Get past 7 days of live prices
        const endDate = new Date();
        const startDate = new Date();
        startDate.setDate(endDate.getDate() - 10); // Fetch 10 to guarantee 7 trading days
 
        const queryOptions = { period1: startDate, period2: endDate, interval: '1d' };
        const result = await yahooFinance.historical(ticker, queryOptions);
 
        if (result && result.length > 0) {
            // Map the financial data to our Farmer Marketplace format
            const records = result.map((record) => {
                return {
                    crop: crop,
                    mandi: mandi,
                    locationName: "Global Market", // Indicating source
                    location: { type: "Point", coordinates: [0, 0] },
                    // Convert standard trading price to a localized ₹ per Quintal estimate
                    pricePerQuintal: parseFloat((record.close * 83.5).toFixed(2)),
                    date: record.date
                };
            });
 
            // Keep only the latest 7 trading days
            const recentRecords = records.slice(-7);
 
            // Prevent duplicating identical dates in DB
            await MandiPrice.deleteMany({ crop, mandi });
 
            await MandiPrice.insertMany(recentRecords);
            console.log(`Successfully fetched Keyless LIVE market data for ${crop}`);
            return recentRecords;
        } else E{
            console.log(`No live trading data found for ${crop}`);
            return [];
        }
    } catch (err) {
        console.error("Error fetching live historical data (Yahoo Finance):", err.message);
        throw err;
    }
};
 
// Emergency fallback if crop isn't traded globally
function populateFallbackData(crop, mandi) {
    const records = [];
    const now = new Date();
    // Base price per crop
    let basePrice = crop.toLowerCase() === 'tomato' ? 1500 : 2000;
 
    for (let i = -7; i < 0; i++) {
        const date = new Date(now);
        date.setDate(now.getDate() + i);
 
        // Add random daily variance
        const variance = Math.floor(Math.random() * 100) - 50;
 
        records.push({
            crop,
            mandi,
            locationName: "Local Sim",
            location: { type: "Point", coordinates: [77.1025, 28.7041] },
            pricePerQuintal: basePrice + variance,
            date: date
        });
 
        basePrice += variance; // trend
    }
 
    // Asynchronously insert to db to be cached
    MandiPrice.deleteMany({ crop, mandi }).then(() => {
        MandiPrice.insertMany(records).catch(console.error);
    });
 
    return records;
}
